# Trufflehog Path Exclusions
# Paths matching these patterns will be excluded from scanning

# Test scripts (may contain test/example credentials)
scripts/notify/test-.*\.sh
scripts/tests?/.*
tests?/fixtures/.*

# Example/template files
.*\.example$
.*\.template$
config/.*\.template$

# Documentation
docs/.*\.md$
README\.md$
CONTRIBUTING\.md$
NOTIFICATIONS\.md$

# CI workflows (use GitHub secrets)
\.github/workflows/.*\.yml$
